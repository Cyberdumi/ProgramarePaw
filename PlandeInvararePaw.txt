    public partial class Form1 : Form
    {
        private List<Companie> companies = new List<Companie>();
        private List<Student> students = new List<Student>();

        public Form1()
        {
            InitializeComponent();
            InitializeCompanies();
            InitializeListView();
        }

        private void InitializeCompanies()
        {
            companies.Add(new Companie(1, "Contoso"));
            companies.Add(new Companie(2, "Fabrikam"));
            companies.Add(new Companie(3, "Adventure Works"));
        }

        private void InitializeListView()
        {
            listViewStudents.View = View.Details;
            listViewStudents.FullRowSelect = true;
            listViewStudents.Columns.Add("Nume", 150);
            listViewStudents.Columns.Add("Data nașterii", 100);
            listViewStudents.Columns.Add("Companie", 150);
        }

        private void btnAdd_Click(object sender, EventArgs e)
        {
            using (var form = new FormStudent(companies))
            {
                if (form.ShowDialog() == DialogResult.OK)
                {
                    students.Add(form.Student);
                    RefreshListView();
                }
            }
        }

        private void RefreshListView()
        {
            listViewStudents.Items.Clear();
            foreach (var s in students)
            {
                var compName = companies
                    .FirstOrDefault(c => c.Id == s.IdCompanie)?.Nume
                    ?? "";
                var item = new ListViewItem(s.Nume);
                item.SubItems.Add(s.DataNasterii.ToShortDateString());
                item.SubItems.Add(compName);
                listViewStudents.Items.Add(item);
            }
        }

        private void listViewStudents_DoubleClick(object sender, EventArgs e)
        {
            if (listViewStudents.SelectedIndices.Count == 0) return;
            int idx = listViewStudents.SelectedIndices[0];
            var orig = students[idx];
            using (var form = new FormStudent(companies, orig))
            {
                if (form.ShowDialog() == DialogResult.OK)
                {
                    students[idx] = form.Student;
                    RefreshListView();
                }
            }
        }

        private void btnDelete_Click(object sender, EventArgs e)
        {
            if (listViewStudents.SelectedIndices.Count == 0) return;
            int idx = listViewStudents.SelectedIndices[0];
            if (MessageBox.Show(
                    "Sigur vrei să ștergi acest student?",
                    "Confirmare", MessageBoxButtons.YesNo, MessageBoxIcon.Question
                ) == DialogResult.Yes)
            {
                students.RemoveAt(idx);
                RefreshListView();
            }
        }

        private void btnCheck_Click(object sender, EventArgs e)
        {
            bool allHave = students.All(s => (bool)s);
            MessageBox.Show(
                allHave
                    ? "Toți studenții au asociată o companie."
                    : "Nu toți studenții au asociată o companie.",
                "Verificare"
            );
        }
    }
}

namespace PracticaApp
{
    public partial class FormStudent : Form
    {
        private List<Companie> companies;
        public Student Student { get; private set; }

        public FormStudent(List<Companie> companies, Student existing = null)
        {
            InitializeComponent();
            this.companies = companies;
            LoadCompanies();

            if (existing != null)
            {
                txtName.Text = existing.Nume;
                dtpBirth.Value = existing.DataNasterii;
                // Select companie sau "Nicio companie"
                var sel = companies.FirstOrDefault(c => c.Id == existing.IdCompanie);
                cbCompanies.SelectedItem = sel ?? cbCompanies.Items[0];
            }
        }

        private void LoadCompanies()
        {
            cbCompanies.Items.Clear();
            cbCompanies.Items.Add(new Companie(0, "Nicio companie"));
            foreach (var c in companies)
                cbCompanies.Items.Add(c);
            cbCompanies.DisplayMember = "Nume";
            cbCompanies.SelectedIndex = 0;
        }

        private void btnOK_Click(object sender, EventArgs e)
        {
            try
            {
                var nume = txtName.Text.Trim();
                if (nume.Length < 2)
                    throw new Exception("Numele trebuie să aibă cel puțin 2 caractere.");

                var dataN = dtpBirth.Value;
                int varsta = DateTime.Now.Year - dataN.Year -
                    (DateTime.Now.DayOfYear < dataN.DayOfYear ? 1 : 0);

                if (varsta <= 20)
                    throw new Exception("Studentul trebuie să aibă mai mult de 20 de ani.");

                var comp = cbCompanies.SelectedItem as Companie;
                Student = new Student(nume, dataN, comp?.Id ?? 0);
                DialogResult = DialogResult.OK;
            }
            catch (Exception ex)
            {
                MessageBox.Show(
                    ex.Message,
                    "Eroare validare",
                    MessageBoxButtons.OK,
                    MessageBoxIcon.Error
                );
                DialogResult = DialogResult.None;
            }
        }
    }

using System;

namespace PracticaApp
{
    public class Student
    {
        public string Nume { get; set; }
        public DateTime DataNasterii { get; set; }
        public int IdCompanie { get; set; }

        public Student(string nume, DateTime dataNasterii, int idCompanie)
        {
            Nume = nume;
            DataNasterii = dataNasterii;
            IdCompanie = idCompanie;
        }

        public int Varsta =>
            DateTime.Now.Year - DataNasterii.Year -
            (DateTime.Now.DayOfYear < DataNasterii.DayOfYear ? 1 : 0);

        public static explicit operator bool(Student s) =>
            s.IdCompanie > 0;
    }
}

using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Windows.Forms;

namespace HrApp
{
    public partial class Form1 : Form
    {
        private List<Department> departamente = new List<Department>();
        private List<Angajat> angajati = new List<Angajat>();
        private int editingIndex = -1;

        public Form1()
        {
            InitializeComponent();
        }

        private void Form1_Load(object sender, EventArgs e)
        {
            // Încarcă departamentele din fișier
            var path = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Departamente.txt");
            if (File.Exists(path))
            {
                foreach (var linie in File.ReadAllLines(path))
                {
                    var parts = linie.Split(',');
                    if (parts.Length == 2 &&
                        int.TryParse(parts[0], out int id) &&
                        !string.IsNullOrWhiteSpace(parts[1]))
                    {
                        departamente.Add(new Department(id, parts[1].Trim()));
                    }
                }
            }
            lbDepartamente.DataSource = departamente;
        }

        private void btnAddUpdate_Click(object sender, EventArgs e)
        {
            // Validări simple
            if (txtNume.Text.Length < 2)
            {
                MessageBox.Show("Numele trebuie să aibă cel puțin 2 caractere.");
                return;
            }
            if (numSalariu.Value <= 0)
            {
                MessageBox.Show("Salariul trebuie să fie > 0.");
                return;
            }

            int? idDept = null;
            if (lbDepartamente.SelectedItem is Department dep)
                idDept = dep.Id;

            if (editingIndex >= 0)
            {
                // Modificare
                var ang = angajati[editingIndex];
                ang.Nume = txtNume.Text.Trim();
                ang.Salariu = numSalariu.Value;
                ang.IdDepartament = idDept;
                editingIndex = -1;
                btnAddUpdate.Text = "Adaugă";
            }
            else
            {
                // Adăugare nouă
                angajati.Add(new Angajat(txtNume.Text.Trim(), numSalariu.Value, idDept));
            }

            RefreshListView();
            ClearInputs();
        }

        private void RefreshListView()
        {
            lvAngajati.Items.Clear();
            foreach (var a in angajati)
            {
                var deptName = departamente.FirstOrDefault(d => d.Id == a.IdDepartament)?.Nume ?? "";
                var item = new ListViewItem(new[] {
                    a.Nume,
                    a.Salariu.ToString("F2"),
                    deptName
                });
                lvAngajati.Items.Add(item);
            }
        }

        private void ClearInputs()
        {
            txtNume.Clear();
            numSalariu.Value = numSalariu.Minimum;
            lbDepartamente.ClearSelected();
        }

        private void deleteToolStripMenuItem_Click(object sender, EventArgs e)
        {
            if (lvAngajati.SelectedIndices.Count > 0)
            {
                int idx = lvAngajati.SelectedIndices[0];
                angajati.RemoveAt(idx);
                RefreshListView();
            }
        }

        private void editToolStripMenuItem_Click(object sender, EventArgs e)
        {
            if (lvAngajati.SelectedIndices.Count > 0)
            {
                editingIndex = lvAngajati.SelectedIndices[0];
                var a = angajati[editingIndex];
                txtNume.Text = a.Nume;
                numSalariu.Value = a.Salariu;
                lbDepartamente.SelectedItem = departamente.FirstOrDefault(d => d.Id == a.IdDepartament);
                btnAddUpdate.Text = "Salvează modificările";
            }
        }
    }
}

                errorProvider.SetError(dateTimePicker1, "Angajatul trebuie sa aiba macar 20 de ani");
                isValid = false;

            }

            if (!isValid)
                return;
            string nume = tbNume.Text;
            
            Companie companieSelectata = comboBox1.SelectedItem as Companie;

            angajatCreat = new Angajat
            {
                Nume = nume,
                DataNasterii = dataNasterii,
                IdCompanie = companieSelectata?.Id ?? 0
            };

            this.DialogResult = DialogResult.OK;
            this.Close();

        }

        public AddForm(List<Companie> companii, Angajat angajatDeEditat) : this(companii)
        {
            tbNume.Text = angajatDeEditat.Nume;
            dateTimePicker1.Value = angajatDeEditat.DataNasterii;

            var companie = companii.FirstOrDefault(c => c.Id == angajatDeEditat.IdCompanie);
            if (companie != null)
                comboBox1.SelectedItem = companie;

            {
                angajati[index] = formEditare.angajatCreat;
                AfisareInListView();
            }
        }

        {
            if (lvAngajati.SelectedItems.Count == 0)
            {

                return;

                "Confirmare stergere",
                MessageBoxButtons.YesNo,
                MessageBoxIcon.Warning

            }

        }

        private void btnAdauga_Click(object sender, EventArgs e)
        {
            var datetime = DateTime.Now.Year;
            bool isValid = true;
            errorProvider.Clear();
            if (string.IsNullOrEmpty(tbModel.Text))
            {

                errorProvider.SetError(tbModel, "Trebuie scrisa marca");
                isValid = false;
            }
            if (!int.TryParse(tbStoc.Text, out int stoc) || stoc <= 0)
            {
                errorProvider.SetError(tbStoc, "Trebuie adaugat un numar");
                isValid = false;
            }
            if (!double.TryParse(tbPret.Text, out double pret) || pret < 0)
            {
                errorProvider.SetError(tbPret, "Trebuie adaugat un pret");
                isValid = false;
            }
            if (!int.TryParse(tbAn.Text, out int an) || an < 1950 || an > datetime)
            {
                errorProvider.SetError(tbPret, "Trebuie adaugat un an");
                isValid = false;
            }

            if (!isValid) return;
            string model = tbModel.Text;
            Producator producator = cbProducator.SelectedItem as Producator;
            smartphoneCreat = new Smartphone
            {
                Id = 0,
                Model = model,
                Pret = pret,
                Stoc = stoc,
                AnAparitie = an,
                IdProducator = producator.Id
            };
            this.DialogResult = DialogResult.OK;
            this.Close();

        }

            cbProducator.SelectedValue = telefon.IdProducator;

            smartphoneCreat = telefon;
        }

using Microsoft.VisualBasic.ApplicationServices;

namespace BiletSmartphone
{
    public partial class MainForm : Form
    {
        private List<Smartphone> smartphones = new List<Smartphone>();
        private List<Producator> producators = new List<Producator>();
        public MainForm()
        {
            InitializeComponent();
            this.Load += MainForm_Load;
            dataGridView1.DoubleClick += Edit_click;

        }

        public void MainForm_Load(object sender, EventArgs e)

        {
            IncarcaProducatori();
            smartphones.Add(new Smartphone(1, "S21", 24, 3000, 2021, 1));
            smartphones.Add(new Smartphone(2, "S22", 29, 3200, 2022, 1));
            smartphones.Add(new Smartphone(3, "16", 35, 3500, 2024, 3));

            Afiseaza();
        }

        private void IncarcaProducatori()
        {
            string caleaFisier = "C:\\Users\\mihai\\source\\repos\\BiletSmartphone\\Producatori.txt";
            string[] linii = File.ReadAllLines(caleaFisier);
            foreach (string linie in linii)
            {
  
                string[] campuri = linie.Split(",");
                if (campuri.Length == 2 &&
                    int.TryParse(campuri[0], out int id) &&
                    !string.IsNullOrWhiteSpace(campuri[1]))
                {
                    producators.Add(new Producator(id, campuri[1].Trim()));
                }
            }
        }

        private void Afiseaza()
        {
            var listaAfisare = smartphones
                .Where(s => producators.Any(p => p.Id == s.IdProducator))
                .Select(s => new
                {
                    s.Id,
                    Producator = producators.First(p => p.Id == s.IdProducator).Denumire,
                    s.Model,
                    s.Pret,
                    s.AnAparitie,
                    s.Stoc,
                }).ToList();

            dataGridView1.DataSource = null;
            dataGridView1.DataSource = smartphones;

        }

        private void btnAdd_Click(object sender, EventArgs e)
        {
            AddForm addForm = new AddForm(producators);
            if (addForm.ShowDialog() == DialogResult.OK)
            {
                Smartphone nou = addForm.smartphoneCreat;
                nou.Id = smartphones.Any() ? smartphones.Max(s => s.Id) + 1 : 1;
                smartphones.Add(nou);
                Afiseaza();
            }
        }

        private void Edit_click(object sender, EventArgs e)
        {
            if (dataGridView1.CurrentRow != null)
            {
                var rand = dataGridView1.CurrentRow.DataBoundItem;

                if (rand is Smartphone smartphoneSelectat)
                {
                    var formEdit = new AddForm(producators);
                    formEdit.PopuleazaDate(smartphoneSelectat);

                    if (formEdit.ShowDialog() == DialogResult.OK)
                    {
                        Smartphone actualizat = formEdit.smartphoneCreat;

                        smartphoneSelectat.Model = actualizat.Model;
                        smartphoneSelectat.Stoc = actualizat.Stoc;
                        smartphoneSelectat.Pret = actualizat.Pret;
                        smartphoneSelectat.AnAparitie = actualizat.AnAparitie;
                        smartphoneSelectat.IdProducator = actualizat.IdProducator;

                        Afiseaza();
                    }
                }
            }
        }

        private void btnDelete_Click(object sender, EventArgs e)
        {
            var index = dataGridView1.CurrentRow.DataBoundItem as Smartphone;
            smartphones.Remove(index);
            Afiseaza();
        }
    }
